Headers المطلوبة
Content-Type: application/json
X-API-Key: VAULT-XXXX-XXXX-XXXX-XXXX
Body (JSON)

  "event": "new_order", // اسم الحدث (مطلوب)
  "status": "success", // success | pending | error
  "source": "website", // مصدر البيانات
  "data": { ... } // أي بيانات إضافية }
Response
{ "ok": true, "id": "uuid-xxxx", "message": "تم الحفظ" }
GET /api/records استرجاع السجلات
GET /api/records?limit=50&status=success

// Query Params:
// limit — عدد السجلات (افتراضي: 50)
// status — فلتر الحالة
// event — فلتر الحدث
// from — تاريخ البداية (ISO)
// to — تاريخ النهاية (ISO)

const API_KEY = 'VAULT-XXXX-XXXX-XXXX';

async function archive(event, data) {
  const res = await fetch('https://zatboctyiagjvnducxzi.supabase.co/rest/v1/vault_records', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'apikey': API_KEY,
      'Authorization': 'Bearer ' + API_KEY
    },
    body: JSON.stringify({ event, payload: data, status: 'success', source: 'website' })
  });
  return res.json();
}
